CREATE TABLE PACIENTE (
DNI VARCHAR(255),
NOMBRE VARCHAR(255),
APELLIDOS VARCHAR(255),
CORREO VARCHAR(255),
TELEFONO INT,
CONSTRAINT PK_PACIENTE PRIMARY KEY (DNI)
)

CREATE TABLE CITA (
IDCITA VARCHAR(255),
IDPACIENTE VARCHAR(255),
URGENTE BIT,
HORA_ENTRADA_ESTIMADA TIME,
HORA_SALIDA_ESTIMADA TIME,
PACIENTE_ACUDIDO BIT,
HORA_ENTRADA_REAL TIME,
HORA_SALIDA_REAL TIME,
FECHA DATE,
CORREO_PACIENTE VARCHAR(255),
TELEFONO_PACIENTE VARCHAR(255),
CONSTRAINT PK_CITA PRIMARY KEY (IDCITA),
CONSTRAINT FK_CITA_PACIENTE FOREIGN KEY (IDPACIENTE) REFERENCES PACIENTE (DNI)
)
